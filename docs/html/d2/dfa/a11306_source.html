31 void handle_key(char c)
32 {
33 if (c == 0) return;
34
35 switch(c)
36 {
37 // PS/2 Keyboard sends '\n' (from keyboard driver)
38 // Serial sends '\r' (from serial driver)
39 case '\r':
40 case '\n':
41 basic::printf("\n");
42
43 m_line_buffer[m_buffer_index] = '\0';
44 if (m_buffer_index > 0) {
45 process_command(m_line_buffer);
46 }
47
48 // Reset for next line
49 m_buffer_index = 0;
50 m_line_buffer[0] = '\0';
51 basic::printf("\nOSOS> "); // Print a new prompt
52 break;
53
54 case 8: // Backspace '\b'
55 if (m_buffer_index > 0){
56 m_buffer_index--;
57 m_line_buffer[m_buffer_index] = '\0';
58 basic::printf("\b \b"); // '\b' only move cursor back in serial terminal (e.g. my Ubuntu terminal), so i am printing ' ' once and then again '\b'
59 }
60 break;
61
62 // Handle printable characters
63 default:
64 if (c >= ' ' && c <= '~' && m_buffer_index < 254)
65 {
66 m_line_buffer[m_buffer_index++] = c;
67 m_line_buffer[m_buffer_index] = '\0';
68 basic::printf("%c", c);
69 }
70 break;
71 }
72 }

77 {
78 // Simple "strcmp"

80 basic::printf("OSOS Kernel Shell. Type 'pci' to list devices.");
