27uint32_t MouseDriver::handleInterrupt(uint32_t esp)
28{
29 // Check if the mouse has sent data
30 uint8_t status = commandPort.read();
31 if(!(status & 0x20)) return esp;
32
33 // Read the next byte of the packet
34 buffer[offset] = dataPort.read();
35 offset = (offset + 1) % 3;
36
37 // When a full 3-byte packet is received
38 if(offset == 0){
39 if(mouseEventHandler == 0) return esp; // Do nothing without a handler
40
41 int8_t delta_x = buffer[1];
42 int8_t delta_y = -buffer[2]; // Y-axis is inverted from the mouse's perspective
43
44 if (delta_x != 0 || delta_y != 0) {
45 mouseEventHandler->onMouseMove(delta_x, delta_y);
46 }
47
48 for(uint8_t i = 0; i < 3; i++) {
49 // Check if the state of button 'i' has changed
50 if((buffer[0] & (1 << i)) != (buttons & (1 << i))) {
51 if(buffer[0] & (1 << i)) {
52 mouseEventHandler->onMouseDown(i + 1); // Left=1, Right=2, Middle=3
53 } else {
54 mouseEventHandler->onMouseUp(i + 1);
55 }
56 }
57 }
58 buttons = buffer[0]; // Save the current button state for the next interrupt
59 }
60
61 return esp;
62}

64void MouseDriver::activate()
65{
66 while(commandPort.read() & 1) dataPort.read();
67 offset = 0;
