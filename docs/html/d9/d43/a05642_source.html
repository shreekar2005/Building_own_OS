10GDT_Row::GDT_Row(uint32_t base, uint32_t limit, uint8_t access_byte, uint8_t gran_byte)
11{
12 this->limit_low = (limit & 0xFFFF);
13 this->base_low = (base & 0xFFFF);
14 this->base_middle = (base >> 16) & 0xFF;
15 this->base_high = (base >> 24) & 0xFF;
16
17 // The upper 4 bits of the 20-bit limit go into the lower 4 bits of the granularity byte.
18 this->granularity = ((limit >> 16) & 0x0F);
19 // The granularity flags (G, D/B, L, AVL) go into the upper 4 bits.
20 this->granularity |= (gran_byte & 0xF0);

50void GDT_Manager::installTable()
51{
52 // A structure that matches the 6-byte format required by the 'lgdt' instruction.
53 struct GDT_Pointer {
54 uint16_t limit;
55 uint32_t base; // Use uint32_t for our i686 (32-bit) target
56 } __attribute__((packed));
57
58 GDT_Pointer gdt_ptr;
59 gdt_ptr.limit = this->limit;
60 gdt_ptr.base = this->base;
61
62 asm volatile(
63 // Load the GDT_Pointer structure directly from its memory location.
64 "lgdt %0\n\t"
65
66 // Reload all data segment registers with the kernel data selector.
67 "mov $0x10, %%ax\n\t" // 0x10 is the selector for kernel_DS
68 "mov %%ax, %%ds\n\t"
69 "mov %%ax, %%es\n\t"
70 "mov %%ax, %%fs\n\t"
71 "mov %%ax, %%gs\n\t"
72 "mov %%ax, %%ss\n\t"
73
74 // Perform a 32-bit far jump to reload the CS register.
75 "jmp $0x08, $flush_cs\n\t" // 0x08 is the selector for kernel_CS
76 "flush_cs:\n\t"
77 : // No output operands
78 : "m"(gdt_ptr) // "m" will put address of `gdt_ptr` in place of `%0`
79 : "memory", "eax" // Clobbered registers
80 );
81 basic::printf("GDT Installed\n");
82}

85void GDT_Manager::printLoadedTable()
86{
87 struct GDT_Pointer {
88 uint16_t limit;
89 uint32_t base;
90 } __attribute__((packed));
91
92 GDT_Pointer gdtr;
93 // Store the current GDT register contents into our struct.
94 asm volatile("sgdt %0" : "=m"(gdtr));
95
96 basic::printf("---\n");
97 basic::printf("INFO about : Currently Loaded GDT\n");
98 basic::printf("Base Address: %#x\n", gdtr.base);
99 basic::printf("Limit: %#x (%d bytes)\n", gdtr.limit, gdtr.limit);
100 basic::printf("Entries: %d\n", (gdtr.limit + 1) / sizeof(GDT_Row));
101 basic::printf("---\n");
102
103 GDT_Row* gdt_entries = (GDT_Row*)gdtr.base;
104 int num_entries = (gdtr.limit + 1) / sizeof(GDT_Row);
105
106 for (int i = 0; i < num_entries; i++) {
107 GDT_Row entry = gdt_entries[i];
108
109 // Reconstruct the base and limit from the scattered fields.
110 uint32_t base = entry.base_high << 24 | entry.base_middle << 16 | entry.base_low;
111 uint32_t limit = (entry.granularity & 0x0F) << 16 | entry.limit_low;
112
113 // If the granularity bit is set, the limit is in 4 KiB pages.
114 if ((entry.granularity & 0x80) != 0) {
115 limit = (limit << 12) | 0xFFF;
116 }
117
118 basic::printf("GDT Entry %d: Base=%p, Limit=%#x, Access=%#x, Granularity=%#x\n",
119 i, (void*)(uintptr_t)base, limit, entry.access, entry.granularity);
120 }
121 basic::printf("---\n");
122}
