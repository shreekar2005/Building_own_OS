38InterruptManager::InterruptManager(essential::GDT_Manager* gdt)
39{
40 // ICW1: Start Initialization Sequence. Both PICs are told to listen for 3 more bytes of config data.
41 picMasterCommand.write(0x11);
42 picSlaveCommand.write(0x11);
43
44 // ICW2: Vector Remapping.
45 // Master is told to remap its IRQs (0-7) to CPU vectors 0x20-0x27.
46 picMasterData.write(0x20);
47 // Slave is told to remap its IRQs (8-15) to CPU vectors 0x28-0x2F.
48 picSlaveData.write(0x28);
49
50 // ICW3: Chaining Configuration.
51 // Master is told a slave is connected on its IRQ 2 line (0x04 = bit 2 set).
52 picMasterData.write(0x04);
53 // Slave is told its identity is 2.
54 picSlaveData.write(0x02);
55
56 // ICW4: Environment Information.
57 // Both are told to operate in standard "8086/88" mode.
58 picMasterData.write(0x01);
59 picSlaveData.write(0x01);
60
61 // OCW1: Interrupt Masking.
62 // This is the final step, enabling all interrupts by writing a mask of all zeros.
63 picMasterData.write(0x00);
64 picSlaveData.write(0x00);
65
66 base=(uintptr_t)&interruptDescriptorTable;
67 limit = sizeof(interruptDescriptorTable) - 1;
68
69 uint16_t kernelCSselectorOffset = gdt->kernel_CS_selector();
70 uint8_t IDT_INTERRUPT_GATE=0xE;
71 for(int i=0; i<256; i++){
72 setIDTEntry(i, kernelCSselectorOffset, &ignoreInterrupt, 0, IDT_INTERRUPT_GATE);
73 }
74
75 // Set handlers for hardware interrupts
76 setIDTEntry(0x20, kernelCSselectorOffset, &handleIRQ0x00, 0, IDT_INTERRUPT_GATE); // Timer
77 setIDTEntry(0x21, kernelCSselectorOffset, &handleIRQ0x01, 0, IDT_INTERRUPT_GATE); // Keyboard
78 setIDTEntry(0x2C, kernelCSselectorOffset, &handleIRQ0x0C, 0, IDT_INTERRUPT_GATE); // PS/2 Mouse
79}

95 uint8_t DescriptorType)

198uintptr_t InterruptManager::handleInterrupt(uint8_t interruptNumber, uintptr_t esp)
199{
200 // Use the global pointer "installed_interrupt_manager" to access the current interrupt manager
201 if(installed_interrupt_manager->handlers[interruptNumber]!=nullptr){
202 esp = installed_interrupt_manager->handlers[interruptNumber]->handleInterrupt(esp);
203 }
204
205 else if(interruptNumber!=0x20){ //0x20 is Hardware Timer Interrupt
206 basic::printf("UNHANDLED INTERRUPT %#hx\n",interruptNumber);
207 }
208
209 // Hardware interrupts must still be acknowledged to the PIC
210 if (0x20 <=interruptNumber && interruptNumber <= 0x2F) {
211 picMasterCommand.write(0x20);
212 if (0x28 <= interruptNumber) picSlaveCommand.write(0x20);
213 }
214
215 return esp;
216}
