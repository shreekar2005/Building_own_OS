31 void handle_key(char c)
32 {
33 if (c == 0) return;
34
35 switch(c)
36 {
37 // PS/2 Keyboard sends '\n' (from keyboard driver)
38 // Serial sends '\r' (from serial driver)
39 case '\r':
40 case '\n':
41 basic::printf("\n");
42 m_line_buffer[m_buffer_index] = '\0';
43 if (m_buffer_index > 0) {
44 process_command(m_line_buffer);
45 }
46 // Reset for next line
47 m_buffer_index = 0;
48 m_line_buffer[0] = '\0';
49 basic::printf("\nOSOS> "); // Print a new prompt
50 break;
51
52 case 8: // Backspace '\b'
53 if (m_buffer_index > 0){
54 m_buffer_index--;
55 m_line_buffer[m_buffer_index] = '\0';
56 basic::printf("\b \b"); // '\b' only move cursor back in serial terminal (e.g. my Ubuntu terminal), so i am printing ' ' once and then again '\b'
57 }
58 break;
59
60 // Handle printable characters
61 default:
62 if (c >= ' ' && c <= '~' && m_buffer_index < 254)
63 {
64 m_line_buffer[m_buffer_index++] = c;
65 m_line_buffer[m_buffer_index] = '\0';
66 basic::printf("%c", c);
67 }
68 break;
69 }
70 }

74 void process_command(const char* command)
75 {
76 // Simple "strcmp"
77 if (command[0] == 'h' && command[1] == 'e' && command[2] == 'l' && command[3] == 'p' && command[4] == '\0') {
78 basic::printf("OSOS Kernel Shell. Type 'pci' to list devices.");
79 }
80 else if (command[0] == 'p' && command[1] == 'c' && command[2] == 'i' && command[3] == '\0') {
81 basic::printf("Listing PCI devices (This is just fake command hehe)...");
82 }
83 else {
