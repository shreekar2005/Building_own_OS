22uint32_t PCI_Controller::getAddress(uint8_t bus, uint8_t device, uint8_t function, uint8_t registerOffset)
23{
24 // Casts to uint32_t for bitwise operations
25 uint32_t lbus = (uint32_t)bus;
26 uint32_t ldevice = (uint32_t)device;
27 uint32_t lfunction = (uint32_t)function;
28
29 // Create configuration address
30 // Formula: (1 << 31) | (bus << 16) | (device << 11) | (function << 8) | (registerOffset & 0xFC)
31 return 0x80000000
32 | (lbus << 16)
33 | (ldevice << 11)
34 | (lfunction << 8)
35 | (registerOffset & 0xFC); // Mask to get 32-bit aligned offset
36}

54uint16_t PCI_Controller::read16(uint8_t bus, uint8_t device, uint8_t function, uint8_t registerOffset)
55{
56 // Read the full 32-bit DWORD that contains the desired word
57 uint32_t dword = read32(bus, device, function, registerOffset);
58
59 // Calculate the shift (0 or 16 bits)
60 // (registerOffset % 4) will be 0 or 2 for a 16-bit aligned read
61 uint32_t shift = (8 * (registerOffset % 4));
62
63 // Shift the desired word to the low bits and cast to uint16_t
64 return (uint16_t)(dword >> shift);

91void PCI_Controller::write16(uint8_t bus, uint8_t device, uint8_t function, uint8_t registerOffset, uint16_t value)
92{
93 // This must be a Read-Modify-Write to avoid corrupting adjacent bytes.
94
95 // 1. Read the current 32-bit DWORD
96 uint32_t dword = read32(bus, device, function, registerOffset);
97
98 // 2. Modify
99 uint32_t shift = (8 * (registerOffset % 4)); // 0 or 16
100 uint32_t mask = 0xFFFF << shift; // Mask for the 16 bits
101
102 dword &= ~mask; // Clear the old 16 bits
103 dword |= ((uint32_t)value << shift); // Set the new 16 bits
104
105 // 3. Write
106 write32(bus, device, function, registerOffset, dword);
107}

141PCI_DeviceDescriptor PCI_Controller::getDeviceDescriptor(uint8_t bus, uint8_t device, uint8_t function)
142{
143 PCI_DeviceDescriptor result;
144 result.bus = bus;
145 result.device = device;
146 result.function = function;
147
148 // --- Identification ---
149 result.vendorId = read16(bus, device, function, 0x00);
150 result.deviceId = read16(bus, device, function, 0x02);
151
152 // --- Classification ---
153 result.classId = read8(bus, device, function, 0x0B);
154 result.subclassId = read8(bus, device, function, 0x0A);
155 result.interfaceId = read8(bus, device, function, 0x09);
156 result.revision = read8(bus, device, function, 0x08);
157
158 // --- Type & Interrupt ---
159 result.headerType = read8(bus, device, function, 0x0E);
160 result.interrupt = read8(bus, device, function, 0x3C);
161
162 // --- Base Address Registers (BARs) ---
163 // These are read as full 32-bit values
164 result.bar0 = read32(bus, device, function, 0x10);
165 result.bar1 = read32(bus, device, function, 0x14);
166 result.bar2 = read32(bus, device, function, 0x18);
167 result.bar3 = read32(bus, device, function, 0x1C);
168 result.bar4 = read32(bus, device, function, 0x20);
169 result.bar5 = read32(bus, device, function, 0x24);
170
171 return result;
172}

175void PCI_Controller::scanBus(uint8_t bus, driver::DriverManager* driver_manager)
176{
177 for(int device=0; device<32; device++)
178 {
179 // Use the new signature with uint8_t
180 int numFunctions = deviceHasFunctions(bus, (uint8_t)device) ? 8 : 1;
181
182 for(int function=0; function<numFunctions; function++)
183 {
184 // Use the new signature with uint8_t
185 PCI_DeviceDescriptor dev = getDeviceDescriptor(bus, (uint8_t)device, (uint8_t)function);
186
187 // Check if device exists
188 if(dev.vendorId == 0x0000 || dev.vendorId == 0xFFFF)
189 continue;
190
191 // 1. Process the device (print info and pass to driver manager)
192 basic::printf("PCI BUS %#3x, DEVICE %#3x, FUNCTION %#3x ", bus, device, function);
193 basic::printf("= VENDOR %#5hx, DEVICE %#5hx (CLASS %#3x, SUB %#3x)\n",
194 dev.vendorId, dev.deviceId, dev.classId, dev.subclassId);
195
196 // driver_manager->registerDevice(dev); // TODO: Uncomment when ready
197
198 // 2. Check if it's a PCI-to-PCI bridge
199 // Class 0x06 = Bridge, Subclass 0x04 = PCI-to-PCI Bridge
200 // We can also check (dev.headerType & 0x7F) == 0x01
201 if(dev.classId == 0x06 && dev.subclassId == 0x04)
