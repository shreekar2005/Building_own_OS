54 void handle_key(char c)
55 {
56 if (c == 0) return;
57
58 switch(c)
59 {
60 // PS/2 Keyboard sends '\n' (from keyboard driver)
61 // Serial sends '\r' (from serial driver)
62 case '\r':
63 case '\n':
64 basic::printf("\n");
65 m_line_buffer[m_buffer_index] = '\0';
66 if (m_buffer_index > 0) {
67 process_command(m_line_buffer);
68 }
69 // Reset for next line
70 m_buffer_index = 0;
71 m_line_buffer[0] = '\0';
72 basic::printf("\nOSOS> "); // Print a new prompt
73 break;
74
75 case 8: // Backspace '\b'
76 if (m_buffer_index > 0){
77 m_buffer_index--;
78 m_line_buffer[m_buffer_index] = '\0';
79 basic::printf("\b \b"); // '\b' only move cursor back in serial terminal (e.g. my Ubuntu terminal), so i am printing ' ' once and then again '\b'
80 }
81 break;
82
83 // Handle printable characters
84 default:
85 if (c >= ' ' && c <= '~' && m_buffer_index < 254)
86 {
87 m_line_buffer[m_buffer_index++] = c;
88 m_line_buffer[m_buffer_index] = '\0';
89 basic::printf("%c", c);
90 }
91 break;
92 }

97 void process_command(const char* command)
98 {
99 if (basic::strcmp(command, "help") == 0) {
100 basic::printf("OSOS Kernel Shell Help\n\
101help : list commands\n\
102clear : clear shell\n\
103lsmem : print memory map provided by grub\n\
104checkmem : check how much memory is free (in KB)\n\
105checkheap : examine kernel heap\n\
106lspagedir : check active page directory entries for kernel\n\
107lspci : list PCI devices\n\
108task<i> : start ith task(thread) from OSOS shell\n\
109numtasks : to see number of tasks in shell list\n");
110 }
111 else if(basic::strcmp(command, "clear") ==0){
112 basic::clearScreen();
113 }
114 else if (basic::strcmp(command, "lsmem") == 0){
115 memory::printMemoryMap(mbi);
116 }
117 else if (basic::strcmp(command, "lspci") == 0){
118 basic::printf("Listing PCI devices...\n");
119 pciController.scanBus(0);
120 }
121 else if (basic::strncmp(command, "task", 4) == 0){
122 char* tasknumstr=(char*)command+4;
123 int tasknum = basic::stoi(tasknumstr) - 1;
124 if(tasknum<0 || tasknum>=numShellThreads)
125 basic::printf("task number %d not present in task list of OSOS shell :(", tasknum+1);
126 else{
127 basic::printf("Starting task number %d...\n", tasknum+1);
128 if (osos_ThreadManager_ptr->addThread(shell_threads_ptr[tasknum])==false) {
129 basic::printf("Error: failed to start task%d.\n", tasknum+1);
130 }
131 }
132 }
133 else if(basic::strcmp(command, "numtasks") ==0){
134 basic::printf("Number of tasks in shell list : %d\n", numShellThreads);
135 }
136 else if(basic::strcmp(command, "checkmem") ==0){
137 basic::printf("Total Free Memory : %d KB\n",memory::PhysicalMemoryManager::get_free_memory_kb());
138 }
139 else if(basic::strcmp(command, "checkheap") ==0){
140 memory::kernel_heap.printHeapInfo();
141 }
142 else if(basic::strcmp(command, "lspagedir") ==0){
143 memory::PagingManager::printPageDirectory();
144 }
145 else {
146 basic::printf("Unknown command: '%s'\n", command);
147 }
148 }
