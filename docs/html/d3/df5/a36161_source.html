28uint32_t MouseDriver::handleInterrupt(uint32_t esp)
29{
30 // Check if the mouse has sent data
31 uint8_t status = commandPort.read();
32 if(!(status & 0x20)) return esp;
33
34 // Read the next byte of the packet
35 buffer[offset] = dataPort.read();
36 offset = (offset + 1) % 3;
37
38 // When a full 3-byte packet is received
39 if(offset == 0){
40 if(mouseEventHandler == 0) return esp; // Do nothing without a handler
41
42 int8_t delta_x = buffer[1];
43 int8_t delta_y = -buffer[2]; // Y-axis is inverted from the mouse's perspective
44
45 if (delta_x != 0 || delta_y != 0) {
46 mouseEventHandler->onMouseMove(delta_x, delta_y);
47 }
48
49 for(uint8_t i = 0; i < 3; i++) {
50 // Check if the state of button 'i' has changed
51 if((buffer[0] & (1 << i)) != (buttons & (1 << i))) {
52 if(buffer[0] & (1 << i)) {
53 mouseEventHandler->onMouseDown(i + 1); // Left=1, Right=2, Middle=3
54 } else {
55 mouseEventHandler->onMouseUp(i + 1);
56 }
57 }
58 }
59 buttons = buffer[0]; // Save the current button state for the next interrupt
60 }
61
62 return esp;
63}
