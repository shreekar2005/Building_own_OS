40 void handle_key(char c)
41 {
42 if (c == 0) return;
43
44 switch(c)
45 {
46 // PS/2 Keyboard sends '\n' (from keyboard driver)
47 // Serial sends '\r' (from serial driver)
48 case '\r':
49 case '\n':
50 basic::printf("\n");
51 m_line_buffer[m_buffer_index] = '\0';
52 if (m_buffer_index > 0) {
53 process_command(m_line_buffer);
54 }
55 // Reset for next line
56 m_buffer_index = 0;
57 m_line_buffer[0] = '\0';
58 basic::printf("\nOSOS> "); // Print a new prompt
59 break;
60
61 case 8: // Backspace '\b'
62 if (m_buffer_index > 0){
63 m_buffer_index--;
64 m_line_buffer[m_buffer_index] = '\0';
65 basic::printf("\b \b"); // '\b' only move cursor back in serial terminal (e.g. my Ubuntu terminal), so i am printing ' ' once and then again '\b'
66 }
67 break;
68
69 // Handle printable characters
70 default:
71 if (c >= ' ' && c <= '~' && m_buffer_index < 254)
72 {
73 m_line_buffer[m_buffer_index++] = c;
74 m_line_buffer[m_buffer_index] = '\0';
75 basic::printf("%c", c);
76 }
77 break;
78 }

83 void process_command(const char* command)
84 {
85 if (basic::strcmp(command, "help") == 0) {
86 basic::printf("OSOS Kernel Shell\n\
87'lspci' : list PCI devices\n\
88'task1' : start taskA\n\
89'task2' : start taskB");
90 }
91 else if (basic::strcmp(command, "lspci") == 0) {
92 basic::printf("Listing PCI devices...\n");
93 pciController.scanBus(0);
94 }
95 else if (basic::strcmp(command, "task1") == 0) {
96 basic::printf("Starting task1...\n");
97 if (osos_TaskManager_ptr->addTask(task1_ptr)==false) {
98 basic::printf("Error: failed to start task1.\n");
99 }
100 }
101 else if (basic::strcmp(command, "task2") == 0) {
102 basic::printf("Starting task2...\n");
103 if (osos_TaskManager_ptr->addTask(task2_ptr)==false) {
104 basic::printf("Error: failed to start task2.\n");
105 }
106 }
107 else {
108 basic::printf("Unknown command: '%s'", command);
109 }
110 }
