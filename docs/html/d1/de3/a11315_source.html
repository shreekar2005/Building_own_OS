20uint32_t PCI_Controller::getAddress(uint8_t bus, uint8_t device, uint8_t function, uint8_t registerOffset)
21{
22 // Casts to uint32_t for bitwise operations
23 uint32_t lbus = (uint32_t)bus;
24 uint32_t ldevice = (uint32_t)device;
25 uint32_t lfunction = (uint32_t)function;
26
27 // Create configuration address
28 // Formula: (1 << 31) | (bus << 16) | (device << 11) | (function << 8) | (registerOffset & 0xFC)
29 return 0x80000000
30 | (lbus << 16)
31 | (ldevice << 11)
32 | (lfunction << 8)
33 | (registerOffset & 0xFC); // Mask to get 32-bit aligned offset
34}

50uint16_t PCI_Controller::read16(uint8_t bus, uint8_t device, uint8_t function, uint8_t registerOffset)
51{
52 // Read the full 32-bit DWORD that contains the desired word
53 uint32_t dword = read32(bus, device, function, registerOffset);
54
55 // Calculate the shift (0 or 16 bits)
56 // (registerOffset % 4) will be 0 or 2 for a 16-bit aligned read
57 uint32_t shift = (8 * (registerOffset % 4));
58
59 // Shift the desired word to the low bits and cast to uint16_t
60 return (uint16_t)(dword >> shift);
61}

87void PCI_Controller::write16(uint8_t bus, uint8_t device, uint8_t function, uint8_t registerOffset, uint16_t value)
88{
89 // This must be a Read-Modify-Write to avoid corrupting adjacent bytes.
90
91 // Read the current 32-bit DWORD
92 uint32_t dword = read32(bus, device, function, registerOffset);
93
94 // Modify
95 uint32_t shift = (8 * (registerOffset % 4)); // 0 or 16
96 uint32_t mask = 0xFFFF << shift; // Mask for the 16 bits
97
98 dword &= ~mask; // Clear the old 16 bits
99 dword |= ((uint32_t)value << shift); // Set the new 16 bits
100
101 // Write
102 write32(bus, device, function, registerOffset, dword);
103}

163void PCI_Controller::scanBus(uint8_t bus, driver::DriverManager* driver_manager)
164{
165 for(int device=0; device<32; device++)
166 {
167 // Use the new signature with uint8_t
168 int numFunctions = deviceHasFunctions(bus, (uint8_t)device) ? 8 : 1;
169
170 for(int function=0; function<numFunctions; function++)
171 {
172 // Use the new signature with uint8_t
173 PCI_DeviceDescriptor dev = getDeviceDescriptor(bus, (uint8_t)device, (uint8_t)function);
174
175 // Check if device exists
176 if(dev.vendorId == 0x0000 || dev.vendorId == 0xFFFF)
177 continue;
178
179 basic::printf("PCI BUS %#3x, DEVICE %#3x, FUNCTION %#3x ", bus, device, function);
180 basic::printf("= VENDOR %#5hx, DEVICE %#5hx (CLASS %#3x, SUB %#3x)\n",
181 dev.vendorId, dev.deviceId, dev.classId, dev.subclassId);
182
183 // driver_manager->registerDevice(dev); // TODO: Uncomment when ready
184
185 // Check if it's a PCI-to-PCI bridge
186 // Class 0x06 = Bridge, Subclass 0x04 = PCI-to-PCI Bridge
187 // We can also check (dev.headerType & 0x7F) == 0x01
188 if(dev.classId == 0x06 && dev.subclassId == 0x04)
189 {
190 // It's a bridge. Read its secondary bus number. (Register 0x19)
191 uint8_t secondaryBus = read8(bus, (uint8_t)device, (uint8_t)function, 0x19);
192
193 // Recursively scan the bus on the other side
194 if (secondaryBus != bus) {
195 scanBus(secondaryBus, driver_manager);
196 }
197 }
198 }
199 }
