92void PhysicalMemoryManager::init(multiboot_info_t* mbi)
93{
94 // placing it at the address of 'end', ensuring it doesn't overwrite kernel code.
95 memory_bitmap = (uint8_t*)&end;
96
97 // Mark EVERYTHING as used (1) initially.
98 // We only free what Multiboot tells us is actually RAM.
99 for (int i = 0; i < (int)BITMAP_SIZE; i++) {
100 memory_bitmap[i] = 0xFF;
101 }
102
103 // Parse Multiboot Map to unlock available RAM
104 uintptr_t mmap_start = mbi->mmap_addr;
105 uintptr_t mmap_end = mbi->mmap_addr + mbi->mmap_length;
106
107 for (multiboot_memory_map_t *mmap = (multiboot_memory_map_t *)mmap_start;
108 (uintptr_t)mmap < mmap_end;
109 mmap = (multiboot_memory_map_t *)((uintptr_t)mmap + mmap->size + sizeof(mmap->size)))
110 {
111 if (mmap->type == MULTIBOOT_MEMORY_AVAILABLE) {
112 // Calculate which 4KB blocks this region covers
113 uint32_t start_block = mmap->addr / PAGE_SIZE;
114 uint32_t num_blocks = mmap->len / PAGE_SIZE;
115
116 // Calculate total blocks supported by our bitmap size
117 uint32_t max_supported_blocks = BITMAP_SIZE * 8;
118
119 // If memory starts beyond 4GB, ignore it
120 if (start_block >= max_supported_blocks) {
121 continue;
122 }
123
124 // If memory starts inside 4GB but extends beyond it, cap it.
125 if (start_block + num_blocks > max_supported_blocks) {
126 num_blocks = max_supported_blocks - start_block;
127 }
128
129 // Mark them as free
130 for (uint32_t i = 0; i < num_blocks; i++) {
131 mark_free(start_block + i);
132 }
133
134 // Track total system size roughly
135 if (start_block + num_blocks > total_blocks) {
136 total_blocks = start_block + num_blocks;
137 }
138 }
139 }
140
141 // Protect Critical Regions
142 // Mark the first 4MB as USED. This covers:
143 // - Real Mode IVT (0x0)
144 // - BIOS Data Area
145 // - Video Memory (0xB8000)
146 // - Your Kernel Code (loaded at 1MB usually)
147 // - GRUB Modules
148 uint32_t reserved_blocks = (4 * 1024 * 1024) / PAGE_SIZE; // 1024 blocks
149 for (uint32_t i = 0; i < reserved_blocks; i++) {
150 mark_used(i);
151 }
152
153 printf("PMM Initialized. Total Blocks: %d\n", total_blocks);
154}
